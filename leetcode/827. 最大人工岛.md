**解题思路**

- 使用`res`表示最终解

- 首先使用`bfs`求出每个岛屿的面积, 以及每个节点对应的岛屿编号, `res = max(res, 每个岛屿面积)`
- 然后枚举所有的0节点, 计算出其周围四个节点汇总所有不同岛屿节点的面积总和`sum`, `res = max(res, sum)`







**代码实现**

```cc
class Solution {
public:
    #define x first
    #define y second
    static constexpr int dx[4] = {-1, 0, 1, 0};
    static constexpr int dy[4] = {0, 1, 0, -1};
    int cnt = -1;

    int bfs(int x, int y, int n, 
    vector<bool> &visited, vector<int> &id, vector<vector<int>> &grid) {
        queue<pair<int, int>> q;
        q.push({x, y});
        id[x * n + y] = cnt;
        visited[x * n + y] = true;
        int area = 0;
        while (q.size()) {
            auto t = q.front();
            q.pop();
            area ++;
            for (int i = 0; i < 4; i++) {
                int nx = t.x + dx[i];
                int ny = t.y + dy[i];
                if (nx < 0 || ny < 0 || nx >= n || ny >= n) continue;
                if (visited[nx * n + ny] || !grid[nx][ny]) continue;
                q.push({nx, ny});
                id[nx * n + ny] = cnt;
                visited[nx * n + ny] = true;
            }
        }
        return area;
    }

    int largestIsland(vector<vector<int>>& grid) {
        int n = grid.size();
        vector<int> id(n * n, 0);
        vector<bool> visited(n * n, false);
        vector<int> area;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (!visited[i * n + j] && grid[i][j] == 1) {
                    cnt ++;
                    area.push_back(bfs(i, j, n, visited, id, grid));
                }
            }
        }

        int res = 0;
        for (int i = 0; i < area.size(); i++) res = max(res, area[i]);
        for (int i = 0; i < n ;i++) {
            for (int j = 0; j < n ;j++) {
                if (grid[i][j] == 0) {
                    unordered_map<int, int> t;
                    res = max(res, 1);
                    for (int k = 0; k < 4; k++) {
                        int nx = i + dx[k];
                        int ny = j + dy[k];
                        if (nx < 0 || ny < 0 || nx >= n || ny >= n) continue;
                        if (grid[nx][ny] == 0) continue;
                        t[id[nx * n + ny]] = area[id[nx * n + ny]];
                    }
                    int sum = 0;
                    for (auto iter = t.begin(); iter != t.end(); iter ++) {
                        sum += iter->second;
                    }
                    res = max(res, sum + 1);
                }
            }
        }
        return res;
    }
};
```
