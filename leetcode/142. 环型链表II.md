```cc
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        ListNode *f = head, *s = head;
        
        while (true) {
            if (!f || !f->next) return nullptr;
            f = f->next->next;
            s = s->next;
            if (f == s) break ;
        }
        f = head;
        while (true) {
            if (f == s) break;
            f = f->next;
            s = s->next;
        }
        return f;
    }
};
```
